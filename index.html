sensorx_intro.ino
// SensorX Club — first session
// You will build this on day one

#include <DHT.h>
#define SENSOR_PIN 7

DHT dht(SENSOR_PIN, DHT11);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  float temp = dht.readTemperature();
  Serial.print("Temp: ");
  Serial.println(temp);
  delay(2000);
}

// Output: Temp: 69°C ✓
